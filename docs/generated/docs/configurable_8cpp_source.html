35void Configurable::load_configuration() {
36 if (config_path_ == "") {
37 ESP_LOGI(__FILENAME__,
38 "Not loading configuration: no config_path specified: %s",
39 config_path_.c_str());
40 return;
41 }
42 String hash_path = String("/") + Base64Sha1(config_path_);
43
44 const String* filename;
45
46 if (SPIFFS.exists(hash_path)) {
47 filename = &hash_path;
48 ESP_LOGD(__FILENAME__, "Loading configuration for path '%s' from '%s'",
49 config_path_.c_str(), hash_path.c_str());
50 } else if (SPIFFS.exists(hash_path + "\n")) {
51 // Up to SensESP v2.4.0, the config path hash had an accidental newline
52 // appended to it.
53 hash_path += "\n";
54 filename = &hash_path;
55 ESP_LOGD(__FILENAME__, "Loading configuration for path '%s' from '%s'",
56 config_path_.c_str(), hash_path.c_str());
57 } else if (config_path_.length() < 32 && SPIFFS.exists(config_path_)) {
58 // Prior to SensESP v2.1.0, the config path was a plain filename.
59 filename = &config_path_;
60 ESP_LOGD(__FILENAME__, "Loading configuration for path %s",
61 config_path_.c_str());
62 } else {
63 ESP_LOGI(__FILENAME__, "Could not find configuration for path %s",
64 config_path_.c_str());
65 return;
66 }
67
68 File f = SPIFFS.open(*filename, "r");
69 JsonDocument json_doc;
70 auto error = deserializeJson(json_doc, f);
71 if (error) {
72 ESP_LOGW(__FILENAME__, "WARNING: Could not parse configuration for %s",
73 config_path_.c_str());
74 return;
75 } //
76 if (!set_configuration(json_doc.as<JsonObject>())) {
77 ESP_LOGW(__FILENAME__, "WARNING: Could not set configuration for %s",
78 config_path_.c_str());
79 }
80 f.close();

83void Configurable::save_configuration() {
84 if (config_path_ == "") {
85 ESP_LOGI(__FILENAME__,
86 "WARNING: Could not save configuration (config_path not set)");
87 }
88 String hash_path = String("/") + Base64Sha1(config_path_);
89
90 if (config_path_.length() < 32 && SPIFFS.exists(config_path_)) {
91 ESP_LOGD(__FILENAME__, "Deleting legacy configuration file %s",
92 config_path_.c_str());
93 SPIFFS.remove(config_path_);
94 }
95
96 ESP_LOGD(__FILENAME__, "Saving configuration path %s to file %s",
97 config_path_.c_str(), hash_path.c_str());
98
99 JsonDocument json_doc;
100 JsonObject obj = json_doc["root"].to<JsonObject>();
101 get_configuration(obj);
102 File f = SPIFFS.open(hash_path, "w");
103 serializeJson(obj, f);
104 f.close();
