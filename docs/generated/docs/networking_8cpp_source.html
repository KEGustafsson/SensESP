124void Networking::start_client_autoconnect() {
125 String hostname = SensESPBaseApp::get_hostname();
126 WiFi.setHostname(hostname.c_str());
127
128 // set up WiFi in regular STA (client) mode
129 auto reconnect_cb = [this]() {
130 static uint32_t attempt_num = 0;
131 static uint32_t current_config_idx = 0;
132
133 int num_configs = client_settings_.size();
134
135 if (WiFi.status() == WL_CONNECTED) {
136 attempt_num = 0;
137 current_config_idx = 0;
138 return;
139 }
140
141 // First check if any of the client settings are defined
142 if (num_configs == 0) {
143 ESP_LOGW(__FILENAME__,
144 "No client settings defined. Leaving WiFi client disconnected.");
145 return;
146 }
147
148 uint32_t prev_config_idx = current_config_idx;
149
150 ClientSSIDConfig config;
151
152 // Get next valid client config
153 for (current_config_idx = current_config_idx;
154 current_config_idx < prev_config_idx + num_configs;
155 current_config_idx++) {
156 config = client_settings_[current_config_idx % num_configs];
157 if (config.ssid_ != "" && config.password_ != "") {
158 break;
159 }
160 }
161
162 ESP_LOGD(__FILENAME__, "Current client config index: %d",
163 current_config_idx);
164 ESP_LOGD(__FILENAME__, "Attempt number: %d", attempt_num);
165 ESP_LOGD(__FILENAME__, "Config SSID: %s", config.ssid_.c_str());
166
167 // If no valid client config found, leave WiFi client disconnected
168 if (config.ssid_ == "" || config.password_ == "") {
169 ESP_LOGW(
170 __FILENAME__,
171 "No valid client settings found. Leaving WiFi client disconnected.");
172 return;
173 }
174
175 ESP_LOGI(__FILENAME__,
176 "Connecting to wifi SSID %s (connection attempt #%d).",
177 config.ssid_.c_str(), attempt_num);
178
179 if (!config.use_dhcp_) {
180 ESP_LOGI(__FILENAME__, "Using static IP address: %s",
181 config.ip_.toString().c_str());
182 WiFi.config(config.ip_, config.dns_server_, config.gateway_,
183 config.netmask_);
184 }
185 WiFi.begin(config.ssid_.c_str(), config.password_.c_str());
186 attempt_num++;
187 current_config_idx++; // Move to the next config for the next attempt
188 };
189
190 // Perform an initial connection without a delay.
191 reconnect_cb();
192
193 // Launch a separate onRepeat event to (re-)establish WiFi connection.
194 // Connecting is attempted only every 20 s to allow the previous connection
195 // attempt to complete even if the network is slow.
196 SensESPBaseApp::get_event_loop()->onRepeat(20000, reconnect_cb);
197}

220bool Networking::set_configuration(const JsonObject& config) {
221 if (config["hostname"].is<String>()) {
222 // deal with the legacy Json format
223 String hostname = config["hostname"].as<String>();
224 SensESPBaseApp::get()->get_hostname_observable()->set(hostname);
225
226 if (config["ssid"].is<String>()) {
227 String ssid = config["ssid"].as<String>();
228 String password = config["password"].as<String>();
229
230 if (config["ap_mode"].is<String>()) {
231 bool ap_mode;
232 if (config["ap_mode"].as<String>() == "Access Point" ||
233 config["ap_mode"].as<String>() == "Hotspot") {
234 ap_settings_ = {true, ssid, password};
235 } else {
236 ClientSSIDConfig client_settings = {ssid, password};
237 client_settings_.clear();
238 client_settings_.push_back(client_settings);
239 client_enabled_ = true;
240 }
241 }
242 }
243 } else {
244 // Either an empty config or a new-style config
245 if (config["apSettings"].is<JsonVariant>()) {
246 ap_settings_ = AccessPointSettings::from_json(config["apSettings"]);
247 } else {
248 ap_settings_ = AccessPointSettings(true);
249 }
250 if (config["clientSettings"].is<JsonVariant>()) {
251 const JsonObject& client_settings_json = config["clientSettings"];
252 client_enabled_ = client_settings_json["enabled"] | false;
253 client_settings_.clear();
254 const JsonArray& client_settings_json_array =
255 client_settings_json["settings"];
256 for (const JsonObject& cfg_json : client_settings_json_array) {
257 client_settings_.push_back(ClientSSIDConfig::from_json(cfg_json));
258 }
259 if (client_settings_.size() == 0) {
260 client_enabled_ = false;
261 }
262 }
263 }
264 // Fill in the rest of the client settings array with empty configs
265 while (client_settings_.size() < kMaxNumClientConfigs) {
266 client_settings_.push_back(ClientSSIDConfig());
267 }
268
269 return true;
270}

305int16_t Networking::get_wifi_scan_results(
306 std::vector<WiFiNetworkInfo>& ssid_list) {
307 int num_networks = WiFi.scanComplete();
308 if (num_networks == WIFI_SCAN_RUNNING) {
309 return WIFI_SCAN_RUNNING;
310 }
311 if (num_networks == WIFI_SCAN_FAILED) {
312 return WIFI_SCAN_FAILED;
313 }
314 ssid_list.clear();
315 for (int i = 0; i < num_networks; i++) {
316 WiFiNetworkInfo info(WiFi.SSID(i), WiFi.RSSI(i), WiFi.encryptionType(i),
317 WiFi.BSSID(i), WiFi.channel(i));
318 ssid_list.push_back(info);
319 }
320
321 return num_networks;
322}
