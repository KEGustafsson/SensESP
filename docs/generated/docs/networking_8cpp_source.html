122void Networking::start_client_autoconnect() {
123 String hostname = SensESPBaseApp::get_hostname();
124 WiFi.setHostname(hostname.c_str());
125
126 // set up WiFi in regular STA (client) mode
127 auto reconnect_cb = [this]() {
128 static uint32_t attempt_num = 0;
129 static uint32_t current_config_idx = 0;
130
131 int num_configs = client_settings_.size();
132
133 if (WiFi.status() == WL_CONNECTED) {
134 attempt_num = 0;
135 current_config_idx = 0;
136 return;
137 }
138
139 // First check if any of the client settings are defined
140 if (num_configs == 0) {
141 ESP_LOGW(__FILENAME__,
142 "No client settings defined. Leaving WiFi client disconnected.");
143 return;
144 }
145
146 uint32_t prev_config_idx = current_config_idx;
147
148 ClientSSIDConfig config;
149
150 // Get next valid client config
151 for (current_config_idx = current_config_idx;
152 current_config_idx < prev_config_idx + num_configs;
153 current_config_idx++) {
154 config = client_settings_[current_config_idx % num_configs];
155 if (config.ssid_ != "" && config.password_ != "") {
156 break;
157 }
158 }
159
160 ESP_LOGD(__FILENAME__, "Current client config index: %d", current_config_idx);
161 ESP_LOGD(__FILENAME__, "Attempt number: %d", attempt_num);
162 ESP_LOGD(__FILENAME__, "Config SSID: %s", config.ssid_.c_str());
163
164 // If no valid client config found, leave WiFi client disconnected
165 if (config.ssid_ == "" || config.password_ == "") {
166 ESP_LOGW(
167 __FILENAME__,
168 "No valid client settings found. Leaving WiFi client disconnected.");
169 return;
170 }
171
172 ESP_LOGI(__FILENAME__, "Connecting to wifi SSID %s (connection attempt #%d).",
173 config.ssid_.c_str(), attempt_num);
174
175 if (!config.use_dhcp_) {
176 ESP_LOGI(__FILENAME__, "Using static IP address: %s",
177 config.ip_.toString().c_str());
178 WiFi.config(config.ip_, config.dns_server_, config.gateway_,
179 config.netmask_);
180 }
181 WiFi.begin(config.ssid_.c_str(), config.password_.c_str());
182 attempt_num++;
183 current_config_idx++; // Move to the next config for the next attempt
184 };
185
186 // Perform an initial connection without a delay.
187 reconnect_cb();
188
189 // Launch a separate onRepeat reaction to (re-)establish WiFi connection.
190 // Connecting is attempted only every 20 s to allow the previous connection
191 // attempt to complete even if the network is slow.
192 ReactESP::app->onRepeat(20000, reconnect_cb);
193}

216bool Networking::set_configuration(const JsonObject& config) {
217 if (config.containsKey("hostname")) {
218 // deal with the legacy Json format
219 String hostname = config["hostname"].as<String>();
220 SensESPBaseApp::get()->get_hostname_observable()->set(hostname);
221
222 if (config.containsKey("ssid")) {
223 String ssid = config["ssid"].as<String>();
224 String password = config["password"].as<String>();
225
226 if (config.containsKey("ap_mode")) {
227 bool ap_mode;
228 if (config["ap_mode"].as<String>() == "Access Point" ||
229 config["ap_mode"].as<String>() == "Hotspot") {
230 ap_settings_ = {true, ssid, password};
231 } else {
232 ClientSSIDConfig client_settings = {ssid, password};
233 client_settings_.clear();
234 client_settings_.push_back(client_settings);
235 client_enabled_ = true;
236 }
237 }
238 }
239 } else {
240 // Either an empty config or a new-style config
241 if (config.containsKey("apSettings")) {
242 ap_settings_ = AccessPointSettings::from_json(config["apSettings"]);
243 } else {
244 ap_settings_ = AccessPointSettings(true);
245 }
246 if (config.containsKey("clientSettings")) {
247 const JsonObject& client_settings_json = config["clientSettings"];
248 client_enabled_ = client_settings_json["enabled"] | false;
249 client_settings_.clear();
250 const JsonArray& client_settings_json_array =
251 client_settings_json["settings"];
252 for (const JsonObject& cfg_json : client_settings_json_array) {
253 client_settings_.push_back(ClientSSIDConfig::from_json(cfg_json));
254 }
255 if (client_settings_.size() == 0) {
256 client_enabled_ = false;
257 }
258 }
259 }
260 // Fill in the rest of the client settings array with empty configs
261 while (client_settings_.size() < kMaxNumClientConfigs) {
262 client_settings_.push_back(ClientSSIDConfig());
263 }
264
265 return true;
266}

289 if (WiFi.status() != WL_CONNECTED) {
290 ESP_LOGD(__FILENAME__,

295 int result = WiFi.scanNetworks(true);

297 ESP_LOGE(__FILENAME__, "WiFi scan failed to start");

301int16_t Networking::get_wifi_scan_results(
302 std::vector<WiFiNetworkInfo>& ssid_list) {
303 int num_networks = WiFi.scanComplete();
304 if (num_networks == WIFI_SCAN_RUNNING) {
305 return WIFI_SCAN_RUNNING;
306 }
307 if (num_networks == WIFI_SCAN_FAILED) {
308 return WIFI_SCAN_FAILED;
309 }
310 ssid_list.clear();
311 for (int i = 0; i < num_networks; i++) {
312 WiFiNetworkInfo info(WiFi.SSID(i), WiFi.RSSI(i), WiFi.encryptionType(i),
313 WiFi.BSSID(i), WiFi.channel(i));
314 ssid_list.push_back(info);
315 }
316
317 return num_networks;
318}
