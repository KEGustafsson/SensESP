291template <typename T>
292std::shared_ptr<ConfigItemT<T>> ConfigItem(std::shared_ptr<T> config_object) {
293 ESP_LOGD(__FILENAME__, "Registering ConfigItemT with path %s",
294 config_object->get_config_path().c_str());
295 auto config_item = std::make_shared<ConfigItemT<T>>(config_object);
296 auto base_sptr = std::static_pointer_cast<ConfigItemBase>(config_item);
297 ConfigItemBase::config_items_[config_object->get_config_path()] = base_sptr;
298 return config_item;
299}
